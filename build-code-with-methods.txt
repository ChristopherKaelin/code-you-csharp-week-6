using System;

Console.WriteLine("\n=====");

bool validLength = false;
bool validZeroes = false;
bool validRange = false;
string errorMessage;

string[] ipv4Input = {"107.31.1.5", "255.0.02.255", "555..0.555", "255...255"};
string[] address;

foreach (string ip in ipv4Input) {
    errorMessage ="";
    address = ip.Split(".", StringSplitOptions.RemoveEmptyEntries);

    ValidateLength();
    ValidateZeroes();
    ValidateRange();

    if (validLength && validZeroes && validRange)
    {
        Console.WriteLine($"{ip} is a valid IPv4 address");
    }
    else
    {
        Console.WriteLine($"{ip} is an invalid IPv4 address.\n\t{errorMessage}");
    }
}



//  IP ipAddress consists of 4 numbers
void ValidateLength() {
    if (address.Length != 4) {
        validLength = false;
        errorMessage += "Not enough numbers; ";
        return;
    }
    validLength = true;
}

//  Each ipAddress number has no leading zeroes
void ValidateZeroes() {
    foreach (string number in address)
    {
        if (number.Length > 1 && number.StartsWith("0"))
        {
            validZeroes = false;
            errorMessage += "Number(s) start with 0; ";
            return;
        }
    }
    validZeroes = true;
}

//  Each ipAddress number is in range 0 - 255
void ValidateRange() {
    foreach (string number in address) {
        int value = int.Parse(number);
        if (value < 0 || value > 255) {
            validRange = false;
            errorMessage += "Number(s) outside valid range; ";
            return;
        }
    }

    validRange = true;
}


Console.WriteLine("\n=====");
